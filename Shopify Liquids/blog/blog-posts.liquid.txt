  ============================================================================
  AVRO — BLOG POSTS  (Shopify Horizon section — adjusted from default)
  ----------------------------------------------------------------------------
  WHAT CHANGED vs. the default Horizon blog-posts section:
  1. Added an editable HEADER (eyebrow + heading + intro) with AVRO copy.
     Turn it off with the "Show header" toggle. Leave a field blank to hide it.
  2. Added an EMPTY STATE message for when a blog has no articles yet.
  3. Replaced the "t:" translation-key labels with plain English labels so the
     section reads clearly in the editor.
  KEPT (do not remove):
  - The responsive grid logic that scales the first post into a hero.
  - Horizon theme blocks: {% content_for 'blocks' %} and the per-article
    'static-blog-post-card' block. These render your real articles.
  - Horizon color/typography tokens (var(--color-foreground), color schemes),
    so this section follows your Theme settings > Colors + Typography.

  HOW TO USE
  - This is the "Blog posts" template section. Assign it to your blog template
    (Online Store > Themes > Customize > pick a Blog page).
  - Edit the header text below in the theme editor. Article cards come straight
    from your Shopify blog posts.
  ============================================================================

<script
  src="{{ 'blog-posts-list.js' | asset_url }}"
  type="module"
  fetchpriority="low"
></script>

{% liquid
  if section.settings.background_color != blank
    render 'contrast-override', background_color: section.settings.background_color, section_id: section.id
  endif
%}

<blog-posts-list section-id="{{ section.id }}">
  <div
    class="section-background{% if section.settings.background_color != blank %} color-custom-{{ section.id }}{% endif %}"
  ></div>
  <div
    class="
      section
      {% if section.settings.background_color != blank %}color-custom-{{ section.id }}{% endif %}
      blog-posts
      spacing-style
      size-style
    "
    data-testid="blog-posts"
    style="{% render 'spacing-style', settings: section.settings %}"
  >
    {% if section.settings.show_header %}
      <header class="avro-blog-header avro-blog-header--{{ section.settings.header_alignment }}">
        {% if section.settings.eyebrow != blank %}
          <p class="avro-blog-header__eyebrow">{{ section.settings.eyebrow }}</p>
        {% endif %}
        {% if section.settings.heading != blank %}
          <h1 class="avro-blog-header__heading">{{ section.settings.heading }}</h1>
        {% endif %}
        {% if section.settings.intro != blank %}
          <div class="avro-blog-header__intro">{{ section.settings.intro }}</div>
        {% endif %}
      </header>
    {% endif %}

    {% content_for 'blocks' %}

    {% if blog.articles.size > 0 %}
      <div
        ref="grid"
        class="blog-posts-container"
      >
        {% liquid
          assign items_on_page = blog.articles.size
        %}
        {% for article in blog.articles %}
          {% liquid
            # Defaults for compact layout
            assign col_span = 2
            assign scale = 0.6
            assign is_horizontal = false

            case items_on_page
              when 1
                # Single post: full width hero
                assign col_span = 6
                assign scale = 1
                assign is_horizontal = true
              when 2
                # Two posts: both standard side by side
                assign col_span = 3
                assign scale = 0.8
              when 3
                # Three posts: hero + 2 standard
                if forloop.index == 1
                  assign col_span = 6
                  assign scale = 1
                  assign is_horizontal = true
                else
                  assign col_span = 3
                  assign scale = 0.8
                endif
              when 4
                # Four posts: all standard in 2x2 grid
                assign col_span = 3
                assign scale = 0.8
              when 5
                # Five posts: hero + 4 standard
                if forloop.index == 1
                  assign col_span = 6
                  assign scale = 1
                  assign is_horizontal = true
                else
                  assign col_span = 3
                  assign scale = 0.8
                endif
              else
                # 6+ posts: hero, 2 standard, rest compact
                if forloop.index == 1
                  assign col_span = 6
                  assign scale = 1
                  assign is_horizontal = true
                elsif forloop.index <= 3
                  assign col_span = 3
                  assign scale = 0.8
                else
                  assign col_span = 2
                  assign scale = 0.6
                endif
            endcase
          %}
          {% style %}
            [data-blog-index="{{ section.id }}-{{ forloop.index }}"] {
              --col-span: {{ col_span }};
              --blog-post-card-scale: {{ scale }};
            }

            @media screen and (max-width: 749px) {
              [data-blog-index="{{ section.id }}-{{ forloop.index }}"] {
                --col-span: 6;
                --blog-post-card-scale: 0.5;
              }
            }
          {% endstyle %}
          <div
            class="blog-post-item{% if is_horizontal %} blog-post-item--horizontal{% endif %}"
            data-testid="blog-post-item"
            data-blog-index="{{ section.id }}-{{ forloop.index }}"
          >
            {% content_for 'block', id: 'static-blog-post-card', type: '_blog-post-card', article: article %}
          </div>
        {% endfor %}
      </div>
    {% else %}
      <div class="avro-blog-empty">
        <p class="avro-blog-empty__title">{{ section.settings.empty_title }}</p>
        <p class="avro-blog-empty__text">{{ section.settings.empty_text }}</p>
      </div>
    {% endif %}
  </div>
</blog-posts-list>
